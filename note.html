				        //scrollTop 向上滚动数值
			            //offsetHeight 元素高度
			            //setInterval 按指定周期执行
			            //clearInterval 清除执行

			            var area = document.getElementById("con");
			            var iLiHeight = 20;
			            var time;
			            var speed = 50;
			            var delay = 2000;
			            area.innerHTML += area.innerHTML;
			            area.scrollTop = 0;

			            function startMove(){
			                area.scrollTop++;
			                time =  setInterval('scrollUp()',speed);
			            }
			            function scrollUp(){
			                if (area.scrollTop % iLiHeight == 0) {
			                    clearInterval(time);
			                    setTimeout("startMove()",delay);
			                }else{
			                    area.scrollTop++;
			                    if (area.scrollTop >= area.scrollHeight/2) {
			                        area.scrollTop=0;
			                    }
			                }
			            }
			            setTimeout("startMove()",delay);
